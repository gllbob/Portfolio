1- docker run --rm -it -v ${PWD}:/app -w /app node:24 npx @angular/cli@21 new frontend --skip-git

2- new Dockerfile in frontend folder:

` FROM node:24

WORKDIR /app

COPY package*.json ./
RUN npm install

COPY . .

EXPOSE 4200

CMD ["npm", "start"] `

3- new docker-compose.yml in root folder:

` services:
  frontend:
    build: ./frontend
    command: npm start
    ports:
      - "4200:4200"
    develop:
      watch:
        - action: sync
          path: ./frontend
          target: /app
          ignore:
            - node_modules/ `

4- new .dockerignore in root folder:

` node_modules
dist
.git
.gitignore
Dockerfile `

5- docker compose up --build --watch

6- git init

7- git add .

8- git commit -m "Arquivos Iniciais do Projeto Portifolio"

9- git branch -M main

10- git remote add origin git@github.com:gllbob/Portfolio.git

11- git push -u origin main
