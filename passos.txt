1- docker run --rm -it -v ${PWD}:/app -w /app node:24 npx @angular/cli@21 new frontend --skip-git

2- new Dockerfile in frontend folder:

` FROM node:24

WORKDIR /app

COPY package*.json ./
RUN npm install

COPY . .

EXPOSE 4200
CMD ["npx", "ng", "serve", "--host", "0.0.0.0"] `

3- new docker-compose.yml in root folder:

` services:
  frontend:
    build: ./frontend
    volumes:
      - ./frontend:/app
    ports:
      - "4200:4200"
    command: npm start `

4- new .dockerignore in root folder:

` node_modules
dist
.git
.gitignore
Dockerfile `

5- docker compose up

6- git init

7- git add .

8- git commit -m "Arquivos Iniciais do Projeto Portifolio"

9- git branch -M main

10- git remote add origin git@github.com:gllbob/Portifolio.git

11- git push -u origin main
